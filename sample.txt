int factorial(int n)
{
    if (n <= 0)
    {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

int main() {
int x = 5;
int result = factorial(x);
int sum = 0;
char c = 'a';

for (int i = 0; i < x; i = i + 1) {
    sum = sum + i;
}

while (sum > 10) {
    sum = sum - 2;
}

if (!(sum == result)) {
    result = result + 1;
}

return result;

}
